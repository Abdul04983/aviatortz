const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); app.use('/api/auth', require('./routes/auth')); app.use('/api', require('./routes/bet')); // for /bet and /cashout mongoose.connect('mongodb://localhost/aviatortz', { useNewUrlParser: true, useUnifiedTopology: true }); app.listen(5000, () => console.log('Server running on port 5000'));
