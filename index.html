const express = require('express'); const router = express.Router(); const auth = require('../middleware/auth'); const User = require('../models/User'); router.post('/bet', auth, async (req, res) => { const { amount } = req.body; if (!amount || amount <= 0) return res.status(400).json({ message: 'Invalid bet amount' }); try { const user = await User.findById(req.user.id); if (user.balance < amount) return res.status(400).json({ message: 'Insufficient balance' }); user.balance -= amount; await user.save(); res.json({ message: 'Bet placed', balance: user.balance }); } catch (err) { res.status(500).json({ message: 'Server error' }); } }); router.post('/cashout', auth, async (req, res) => { const { winnings } = req.body; if (!winnings || winnings <= 0) return res.status(400).json({ message: 'Invalid winnings' }); try { const user = await User.findById(req.user.id); user.balance += winnings; await user.save(); res.json({ message: 'Cash out successful', balance: user.balance }); } catch (err) { res.status(500).json({ message: 'Server error' }); } }); module.exports = router;
